void setup() {
  Serial.begin(VITESSE_UART);
  initBroches();
   
}

void loop() 
{

static unsigned long int dureeON = SECONDE;

if(digitalRead(BOUTON)){
  while(digitalRead(BOUTON));
  //delay(50); // Key-bounce
  Serial << "La LED est ON pour " << dureeON << " ms secondes "  << endl;
  digitalWrite(LED1, HIGH);
  delay(dureeON);
  digitalWrite(LED1, LOW);
  delay(dureeON-=100);
  }
}

void initBroches(){
   pinMode(LED1, OUTPUT);
   pinMode(LED2, OUTPUT);
   pinMode(BOUTON, INPUT);
}
