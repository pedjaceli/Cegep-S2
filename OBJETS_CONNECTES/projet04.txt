void loop() {
  char buffer[16+1];
  unsigned long currentMillis = millis();
  unsigned int currentTime = (currentMillis / 10);
  int sec = currentTime % 60;
  int min = (currentTime / 60) % 60 ; // ???
  int heure = (currentTime / 3600) % 24  ; // ???
  // D: 05:10:01 ...
  snprintf(buffer, sizeof(buffer), "D: %02d:%02d:%02d", heure, min, sec);
  //lcd.setCursor(0,0);
  //lcd << buffer;
  Serial << buffer << endl;
} // loop
