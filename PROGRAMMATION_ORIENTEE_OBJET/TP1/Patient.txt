namespace TP1
{
    class Patient : Person
    {
        /// <summary>
        /// Constructeur - demande le prénom et le nom à l'utilisateur
        /// </summary>
        public Patient() : base()
        {

        }


        /// <summary>
        /// Constructeur - recrée un patient à partir d'une ligne du fichier de sauvegarde
        /// </summary>
        public Patient(int id, string firstName, string lastName) : base(id, firstName, lastName)
        {

        }


        /// <summary>
        /// Vrai si le patient est assigné à un médecin
        /// </summary>
        public bool HasDoctor
        {
            get { return _doctor != null; }
        }


        /// <summary>
        /// Assigne un médecin à ce patient
        /// </summary>
        public void AssignDoctor(Doctor doctor)
        {
            _doctor = doctor;
        }


        /// <summary>
        /// Le médecin assigné au patient (null si en attente)
        /// </summary>
        public Doctor Doctor
        {
            get { return _doctor; }
        }


        /// <summary>
        /// Vérifie si le nouveau rendez-vous est en conflit avec l'horaire du patient
        /// </summary>
        public bool HasConflict(Appointment appointment)
        {
            return PersonSchedule.HasConflict(appointment);
        }


        /// <summary>
        /// Ajoute un rendez-vous à l'horaire du patient
        /// </summary>
        public void AddAppointment(Appointment appointment)
        {
            PersonSchedule.Add(appointment);
        }


        /// <summary>
        /// Affiche les informations du patient
        /// Format: 1) Sami Melançon, médecin: 2 (Louise Poirier)
        /// </summary>
        public void Print()
        {
            if (_doctor != null)
                Console.WriteLine($"  {Id}) {FullName}, médecin: {_doctor.Id} ({_doctor.FullName})");
            else
                Console.WriteLine($"  {Id}) {FullName}");
        }


        /// <summary>
        /// Retourne la ligne de sauvegarde du patient
        /// </summary>
        public string GetFileLine()
        {
            return FileLine;
        }


        private Doctor _doctor = null;
    }
}
