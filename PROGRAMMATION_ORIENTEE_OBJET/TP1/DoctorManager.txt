namespace TP1
{
    class DoctorManager
    {
        /// <summary>
        /// Constructeur - charge le fichier de sauvegarde si il existe
        /// </summary>
        public DoctorManager()
        {
            try
            {
                using (StreamReader input = new(DoctorFileName))
                {
                    string? line = input.ReadLine();
                    while (line != null)
                    {
                        string[] values = line.Split(Separator);
                        int id = Convert.ToInt32(values[0]);
                        string firstName = values[1];
                        string lastName = values[2];
                        _doctors.Add(new Doctor(id, firstName, lastName));
                        line = input.ReadLine();
                    }
                }
            }
            catch (FileNotFoundException)
            {
                // Le fichier n'existe pas, on démarre sans données
            }
        }


        /// <summary>
        /// Reçoit la référence au gestionnaire de patients
        /// Appelé par le constructeur de PatientManager
        /// </summary>
        public void SetPatientManager(PatientManager patientManager)
        {
            _patientManager = patientManager;
        }


        /// <summary>
        /// Crée un médecin, lui assigne les patients en attente (jusqu'à 3), et l'ajoute à la liste
        /// </summary>
        public void Add()
        {
            Doctor doctor = new();

            // Assigne les patients en attente dans l'ordre d'arrivée
            if (_patientManager != null)
            {
                Patient waiting = _patientManager.GetNextWaitingPatient();
                while (waiting != null && doctor.IsAvailable)
                {
                    doctor.AssignPatient(waiting);
                    waiting.AssignDoctor(doctor);
                    waiting = _patientManager.GetNextWaitingPatient();
                }
            }

            _doctors.Add(doctor);
            Console.WriteLine($"Médecin {doctor.Id} ajouté");
        }


        /// <summary>
        /// Affiche les médecins en 2 catégories : disponibles et non disponibles
        /// </summary>
        public void Print()
        {
            List<Doctor> available = new();
            List<Doctor> full = new();

            foreach (Doctor d in _doctors)
            {
                if (d.IsAvailable) available.Add(d);
                else full.Add(d);
            }

            string header1 = $"Médecins disponibles ({available.Count})";
            Console.WriteLine(header1);
            Console.WriteLine(new string('-', header1.Length));
            foreach (Doctor d in available)
                d.Print();

            Console.WriteLine();

            string header2 = $"Médecins non disponibles ({full.Count})";
            Console.WriteLine(header2);
            Console.WriteLine(new string('-', header2.Length));
            foreach (Doctor d in full)
                d.Print();
        }


        /// <summary>
        /// Sauvegarde les médecins dans le fichier
        /// </summary>
        public void Save()
        {
            using (StreamWriter output = new(DoctorFileName))
            {
                foreach (Doctor d in _doctors)
                    output.WriteLine(d.GetFileLine());
            }
        }


        /// <summary>
        /// Retourne le premier médecin disponible, ou null s'il n'y en a pas
        /// </summary>
        public Doctor FindAvailableDoctor()
        {
            foreach (Doctor d in _doctors)
            {
                if (d.IsAvailable)
                    return d;
            }
            return null;
        }


        /// <summary>
        /// Retourne le médecin ayant le moins de rendez-vous (pour la prise de RDV sans médecin assigné)
        /// </summary>
        public Doctor FindDoctorWithFewestAppointments()
        {
            Doctor minDoctor = null;
            int minCount = int.MaxValue;

            foreach (Doctor d in _doctors)
            {
                if (d.AppointmentCount < minCount)
                {
                    minCount = d.AppointmentCount;
                    minDoctor = d;
                }
            }

            return minDoctor;
        }


        private const string DoctorFileName = "../../../medecins.txt";
        private const char Separator = ';';

        private List<Doctor> _doctors = new();
        private PatientManager _patientManager = null;
    }
}
