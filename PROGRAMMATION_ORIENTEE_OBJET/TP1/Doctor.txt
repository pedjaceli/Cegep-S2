namespace TP1
{
    class Doctor : Person
    {
        /// <summary>
        /// Constructeur - demande le prénom et le nom à l'utilisateur
        /// </summary>
        public Doctor() : base()
        {

        }


        /// <summary>
        /// Constructeur - recrée un médecin à partir d'une ligne du fichier de sauvegarde
        /// </summary>
        public Doctor(int id, string firstName, string lastName) : base(id, firstName, lastName)
        {

        }


        /// <summary>
        /// Vrai si le médecin peut encore prendre des patients
        /// </summary>
        public bool IsAvailable
        {
            get { return _patients.Count < MaxPatients; }
        }


        /// <summary>
        /// Assigne un patient à ce médecin
        /// </summary>
        public void AssignPatient(Patient patient)
        {
            if (IsAvailable)
            {
                _patients.Add(patient);
            }
        }


        /// <summary>
        /// Nombre de patients assignés à ce médecin
        /// </summary>
        public int PatientCount
        {
            get { return _patients.Count; }
        }


        /// <summary>
        /// Nombre de rendez-vous dans l'horaire du médecin
        /// </summary>
        public int AppointmentCount
        {
            get { return PersonSchedule.AppointmentCount; }
        }


        /// <summary>
        /// Vérifie si le nouveau rendez-vous est en conflit avec l'horaire du médecin
        /// </summary>
        public bool HasConflict(Appointment appointment)
        {
            return PersonSchedule.HasConflict(appointment);
        }


        /// <summary>
        /// Ajoute un rendez-vous à l'horaire du médecin
        /// </summary>
        public void AddAppointment(Appointment appointment)
        {
            PersonSchedule.Add(appointment);
        }


        /// <summary>
        /// Affiche les informations du médecin
        /// Format: 2) Louise Poirier, patients: 1 (Sami Melançon)
        /// </summary>
        public void Print()
        {
            string patientNames = "";
            for (int i = 0; i < _patients.Count; i++)
            {
                if (i > 0) patientNames += ", ";
                patientNames += _patients[i].FullName;
            }
            Console.WriteLine($"  {Id}) {FullName}, patients: {_patients.Count} ({patientNames})");
        }


        /// <summary>
        /// Retourne la ligne de sauvegarde du médecin
        /// </summary>
        public string GetFileLine()
        {
            return FileLine;
        }


        private const int MaxPatients = 3;

        private List<Patient> _patients = new();
    }
}
