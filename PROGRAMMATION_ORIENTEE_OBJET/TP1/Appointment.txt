namespace TP1
{
    class Appointment
    {
        /// <summary>
        /// Constructeur
        /// </summary>
        /// <param name="dateTime">Date et heure du rendez-vous</param>
        /// <param name="duration">Durée en minutes (10, 20 ou 30)</param>
        /// <param name="patient">Le patient concerné</param>
        /// <param name="doctor">Le médecin concerné</param>
        public Appointment(DateTime dateTime, int duration, Patient patient, Doctor doctor)
        {
            _dateTime = dateTime;
            _duration = duration;
            _patient = patient;
            _doctor = doctor;
        }


        /// <summary>
        /// Date et heure de début du rendez-vous
        /// </summary>
        public DateTime DateTime
        {
            get { return _dateTime; }
        }


        /// <summary>
        /// Date et heure de fin du rendez-vous
        /// </summary>
        public DateTime EndTime
        {
            get { return _dateTime.AddMinutes(_duration); }
        }


        /// <summary>
        /// Vérifie si ce rendez-vous est en conflit avec un autre
        /// Deux rendez-vous sont en conflit s'ils se chevauchent
        /// </summary>
        public bool ConflictsWith(Appointment other)
        {
            return _dateTime < other.EndTime && EndTime > other._dateTime;
        }


        /// <summary>
        /// Affiche les informations du rendez-vous
        /// Format: 2026-04-16 12:00:00 (30 minutes) pour patient 1 et médecin 2
        /// </summary>
        public void Print()
        {
            Console.WriteLine($"{_dateTime:yyyy-MM-dd HH:mm:ss} ({_duration} minutes) pour patient {_patient.Id} et médecin {_doctor.Id}");
        }


        // Durées disponibles en minutes
        public const int DurationA = 10;
        public const int DurationB = 20;
        public const int DurationC = 30;

        private readonly DateTime _dateTime;
        private readonly int _duration;
        private readonly Patient _patient;
        private readonly Doctor _doctor;
    }
}
