namespace TP1
{
    class Schedule
    {
        /// <summary>
        /// Constructeur
        /// </summary>
        public Schedule()
        {

        }


        /// <summary>
        /// Vérifie si un nouveau rendez-vous serait en conflit avec un existant
        /// </summary>
        public bool HasConflict(Appointment newAppointment)
        {
            foreach (Appointment appointment in _appointments)
            {
                if (newAppointment.ConflictsWith(appointment))
                {
                    return true;
                }
            }
            return false;
        }


        /// <summary>
        /// Ajoute un rendez-vous en conservant l'ordre chronologique
        /// </summary>
        public void Add(Appointment newAppointment)
        {
            // On cherche la bonne position pour insérer en ordre croissant
            for (int i = 0; i < _appointments.Count; i++)
            {
                if (newAppointment.DateTime < _appointments[i].DateTime)
                {
                    _appointments.Insert(i, newAppointment);
                    return;
                }
            }
            // Si plus grand que tous les existants, on ajoute à la fin
            _appointments.Add(newAppointment);
        }


        /// <summary>
        /// Nombre de rendez-vous dans l'horaire
        /// </summary>
        public int AppointmentCount
        {
            get { return _appointments.Count; }
        }


        /// <summary>
        /// Affiche tous les rendez-vous de l'horaire
        /// </summary>
        public void Print()
        {
            foreach (Appointment appointment in _appointments)
            {
                appointment.Print();
            }
        }


        private List<Appointment> _appointments = new();
    }
}
